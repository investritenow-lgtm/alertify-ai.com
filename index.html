// index.js const express = require("express"); const axios = require("axios"); const Anthropic = require("@anthropic-ai/sdk"); const app = express(); app.use(express.json({ limit: "20mb" })); // increased for image payloads app.use(express.urlencoded({ limit: "20mb", extended: true })); const client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY?.trim() }); const SAFE_BROWSING_KEY = process.env.GOOGLE_SAFE_BROWSING_KEY?.trim(); const SAFE_BROWSING_URL = `https://safebrowsing.googleapis.com/v4/threatMatches:find?key=${SAFE_BROWSING_KEY}`; // ─── Health check ───────────────────────────────────────────── app.get("/health", (req, res) => { res.json({ status: "ok", service: "Alertify.AI Backend" }); }); // ─── URL Safety Check (Google Safe Browsing) ───────────────── app.post("/check-url", async (req, res) => { const { url } = req.body; if (!url) return res.status(400).json({ error: "url is required" }); try { const response = await axios.post(SAFE_BROWSING_URL, { client: { clientId: "alertify-ai", clientVersion: "1.0.0" }, threatInfo: { threatTypes: [ "MALWARE", "SOCIAL_ENGINEERING", "UNWANTED_SOFTWARE", "POTENTIALLY_HARMFUL_APPLICATION", ], platformTypes: ["ANY_PLATFORM"], threatEntryTypes: ["URL"], threatEntries: [{ url }], }, }); const matches = response.data.matches || []; const isThreat = matches.length > 0; const threatType = isThreat ? matches[0].threatType : null; console.log(`🔍 URL check: ${url} → ${isThreat ? "⚠️ THREAT: " + threatType : "✅ Safe"}`); return res.json({ url, isThreat, threatType, source: "google_safe_browsing" }); } catch (err) { console.error("Safe Browsing API error:", err.message); return res.json({ url, isThreat: false, threatType: null, source: "error" }); } }); // ─── AI Threat Analysis (text) ──────────────────────────────── app.post("/analyze", async (req, res) => { const { content, type, pushToken } = req.body; if (!content) return res.status(400).json({ error: "content is required" }); const systemPrompt = `You are a scam detection AI. Analyze the content and determine if it is a scam, phishing attempt, or social engineering attack. IMPORTANT RULES: - Only flag content as a threat if you are 85% or more confident it is malicious - Normal app notifications, promotions from known brands, and standard UI text are NOT threats - In-app popups from legitimate apps (settings prompts, update notices, feature announcements) are NOT threats - A URL alone is not a threat — it must show signs of phishing or malware - Words like "urgent", "free", "verify", "limited time" alone are NOT enough — require specific scam context - If in doubt, do NOT flag it — false positives are worse than false negatives for user trust Respond ONLY in JSON with no markdown: { "isThreat": boolean, "confidence": number (0-100), "title": string, "explanation": string, "severity": "low"|"medium"|"high"|"critical", "category": string } Only set isThreat: true if confidence >= 85.`; try { const message = await client.messages.create({ model: "claude-sonnet-4-20250514", max_tokens: 1024, system: systemPrompt, messages: [ { role: "user", content: `Content to analyze (type: ${type || "unknown"}):\n\n${content}`, }, ], }); const raw = message.content[0].text; const clean = raw.replace(/```json|```/g, "").trim(); const analysis = JSON.parse(clean); if (analysis.isThreat && pushToken) { await sendPushNotification(pushToken, analysis, type); } console.log(`🤖 Analysis: isThreat=${analysis.isThreat}, confidence=${analysis.confidence}`); return res.json({ analysis }); } catch (err) { console.error("Analysis error:", err.message); return res.status(500).json({ error: "Analysis failed" }); } }); // ─── Image Scan (Claude Vision) ─────────────────────────────── app.post("/scan-image", async (req, res) => { const { imageBase64, mediaType, pushToken } = req.body; if (!imageBase64) return res.status(400).json({ error: "imageBase64 is required" }); const validTypes = ["image/jpeg", "image/png", "image/gif", "image/webp"]; const resolvedType = validTypes.includes(mediaType) ? mediaType : "image/jpeg"; try { console.log("📷 Scanning image for scam content..."); const message = await client.messages.create({ model: "claude-sonnet-4-20250514", max_tokens: 1024, system: `You are a scam detection AI specializing in physical mail, letters, and printed documents. Analyze the image and extract any text, then determine if the document is a scam, fraud, or social engineering attempt. Common physical mail scams include: - Fake lottery or sweepstakes winnings - Government impersonation (IRS, SSA, Medicare, DMV) - Fake debt collection notices - Romance scams asking for money transfers - Fake check scams - Prize notification letters - Urgent payment demands with wire transfer or gift card instructions - Too-good-to-be-true investment offers Respond ONLY in JSON with no markdown: { "isThreat": boolean, "confidence": number (0-100), "title": string, "explanation": string, "severity": "low"|"medium"|"high"|"critical", "category": string, "extractedText": string (first 300 chars of text found in image) } Only set isThreat: true if confidence >= 80.`, messages: [ { role: "user", content: [ { type: "image", source: { type: "base64", media_type: resolvedType, data: imageBase64, }, }, { type: "text", text: "Please analyze this document or letter for scam indicators. Extract the text and determine if this is a fraudulent communication.", }, ], }, ], }); const raw = message.content[0].text; const clean = raw.replace(/```json|```/g, "").trim(); const analysis = JSON.parse(clean); if (analysis.isThreat && pushToken) { await sendPushNotification(pushToken, analysis, "document_scan"); } console.log(`📷 Image scan: isThreat=${analysis.isThreat}, confidence=${analysis.confidence}`); return res.json({ analysis }); } catch (err) { console.error("Image scan error:", err.message); return res.status(500).json({ error: "Image scan failed" }); } }); // ─── Direct push notification ───────────────────────────────── app.post("/alert", async (req, res) => { const { pushToken, title, body, type, severity, source } = req.body; if (!pushToken) return res.status(400).json({ error: "pushToken is required" }); try { await sendPushNotification( pushToken, { title, explanation: body, severity, category: type }, source ); return res.json({ success: true }); } catch (err) { return res.status(500).json({ error: err.message }); } }); // ─── Push notification helper ───────────────────────────────── async function sendPushNotification(pushToken, analysis, source) { const payload = { to: pushToken, title: analysis.title || "⚠️ Threat Detected", body: analysis.explanation || "A potential scam was detected", sound: "default", priority: "high", channelId: "default", data: { alertId: Date.now().toString(), type: analysis.category || "unknown", severity: analysis.severity || "medium", source: source || "", }, }; await axios.post("https://exp.host/--/api/v2/push/send", payload, { headers: { "Content-Type": "application/json" }, }); } const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`🚀 Alertify.AI backend running on port ${PORT}`); });
